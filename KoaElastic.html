  if (!q) {

    return;
  }
  
  const result = await client.search({
    index: 'articles',
    body: {
      from: (page - 1) * limit,
      size: limit,
      query: {
        multi_match: {
          query: q,
          fields: ['title^2', 'content']
        }
      }
    }
  });
  
  ctx.body = {
    total: result.hits.total.value,
    page: parseInt(page),
    results: result.hits.hits.map(hit => ({
      id: hit._id,
      score: hit._score,
      ...hit._source
    }))
  };
