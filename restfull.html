  if (!usuario) {
    return res.status(404).json({ 
      success: false, 
      error: 'Usuario no encontrado' 
    });
  }
  res.json({ success: true, data: usuario });

  // Validaci√≥n b√°sica

  usuario.updatedAt = new Date().toISOString();
  
  res.json({ success: true, data: usuario });

  const index = usuarios.findIndex(u => u.id === parseInt(req.params.id));
